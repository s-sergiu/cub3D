78static uint32_t mlx_compile_shader(const char* code, int32_t type)
79{
80 GLuint shader;
81 int32_t success;
82 char infolog[512] = {0};
83
84 if (!code || (shader = glCreateShader(type)) == 0)
85 return (0);
86
87 GLint len = strlen(code);
88 glShaderSource(shader, 1, &code, &len);
89 glCompileShader(shader);
90 glGetShaderiv(shader, GL_COMPILE_STATUS, &success);
91 if (!success)
92 {
93 glGetShaderInfoLog(shader, sizeof(infolog), NULL, infolog);
94 fprintf(stderr, "%s", infolog);
95 return (0);
96 }
97 return (shader);
98}

100static bool mlx_init_render(mlx_t* mlx)
101{
102 uint32_t vshader = 0;
103 uint32_t fshader = 0;
104 char infolog[512] = {0};
105 mlx_ctx_t* mlxctx = mlx->context;
106
107 glfwMakeContextCurrent(mlx->window);
108 glfwSetFramebufferSizeCallback(mlx->window, framebuffer_callback);
109 glfwSetWindowUserPointer(mlx->window, mlx);
110 glfwSwapInterval(MLX_SWAP_INTERVAL);
111
112 // Load all OpenGL function pointers
113 if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress))
114 return (mlx_error(MLX_GLADFAIL));
115
116 if (!(vshader = mlx_compile_shader(vert_shader, GL_VERTEX_SHADER)))
117 return (mlx_error(MLX_VERTFAIL));
118 if (!(fshader = mlx_compile_shader(frag_shader, GL_FRAGMENT_SHADER)))
119 return (mlx_error(MLX_FRAGFAIL));
120 if (!(mlxctx->shaderprogram = glCreateProgram()))
121 return (mlx_error(MLX_SHDRFAIL));
122 glAttachShader(mlxctx->shaderprogram, vshader);
123 glAttachShader(mlxctx->shaderprogram, fshader);
124 glLinkProgram(mlxctx->shaderprogram);
125
126 int32_t success;
127 glGetProgramiv(mlxctx->shaderprogram, GL_LINK_STATUS, &success);
128 if (!success)
129 {
130 glGetProgramInfoLog(mlxctx->shaderprogram, sizeof(infolog), NULL, infolog);
131 fprintf(stderr, "%s", infolog);
132 return (mlx_error(MLX_SHDRFAIL));
133 }
134 glDeleteShader(vshader);
135 glDeleteShader(fshader);
136 glUseProgram(mlxctx->shaderprogram);
137
138 for (size_t i = 0; i < 16; i++)
139 mlxctx->bound_textures[i] = 0;
140
141 return (true);
142}

153mlx_t* mlx_init(int32_t width, int32_t height, const char* title, bool resize)
154{
155 MLX_NONNULL(title);
156 MLX_ASSERT(width > 0, "Window width must be positive");
157 MLX_ASSERT(height > 0, "Window height must be positive");
158
159 bool init;
160 mlx_t* mlx;
161 if (!(init = glfwInit()))
162 return ((void*)mlx_error(MLX_GLFWFAIL));
163 if (!(mlx = calloc(1, sizeof(mlx_t))))
164 return ((void*)mlx_error(MLX_MEMFAIL));
165 if (!(mlx->context = calloc(1, sizeof(mlx_ctx_t))))
166 return (free(mlx), (void*)mlx_error(MLX_MEMFAIL));
167
168 mlx_ctx_t* const mlxctx = mlx->context;
169 mlx->width = width;
170 mlx->height = height;
171 mlxctx->initialWidth = width;
172 mlxctx->initialHeight = height;
173
174 glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
175 glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
176 glfwWindowHint(GLFW_MAXIMIZED, mlx_settings[MLX_MAXIMIZED]);
177 glfwWindowHint(GLFW_DECORATED, mlx_settings[MLX_DECORATED]);
178 glfwWindowHint(GLFW_VISIBLE, !mlx_settings[MLX_HEADLESS]);
179 glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);
180#ifdef __APPLE__
181 glfwWindowHint(GLFW_OPENGL_FORWARD_COMPAT, GL_TRUE);
182#endif
183 glfwWindowHint(GLFW_RESIZABLE, resize);
184 if (!(mlx->window = glfwCreateWindow(width, height, title, mlx_settings[MLX_FULLSCREEN] ? glfwGetPrimaryMonitor() : NULL, NULL)))
185 return (mlx_terminate(mlx), (void*)mlx_error(MLX_WINFAIL));
186 if (!mlx_init_render(mlx) || !mlx_create_buffers(mlx))
187 return (mlx_terminate(mlx), NULL);
188 return (mlx);
189}
