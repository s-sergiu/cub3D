21void mlx_update_matrix(const mlx_t* mlx, int32_t width, int32_t height)
22{
23 const mlx_ctx_t* mlxctx = mlx->context;
24 const float depth = mlxctx->zdepth;
25
30 width = mlx_settings[MLX_STRETCH_IMAGE] ? mlxctx->initialWidth : mlx->width;
31 height = mlx_settings[MLX_STRETCH_IMAGE] ? mlxctx->initialHeight : mlx->height;
32
33 const float matrix[16] = {
34 2.f / width, 0, 0, 0,
35 0, 2.f / -(height), 0, 0,
36 0, 0, -2.f / (depth - -depth), 0,
37 -1, -(height / -height),
38 -((depth + -depth) / (depth - -depth)), 1
39 };
40
41 glUniformMatrix4fv(glGetUniformLocation(mlxctx->shaderprogram, "ProjMatrix"), 1, GL_FALSE, matrix);
42}
